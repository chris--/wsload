1 /**
  2  * Module dependencies
  3  */
  4 
  5 var mongo = require('mongodb');
  6 
  7 /** 
  8  * Expose `MongoAdapter`
  9  */
 10 
 11 module.exports = MongoAdapter;
 12 
 13 /**
 14  * Initialize a new `MongoAdapter`.
 15  *
 16  * Connects to a MongoDB on localhost
 17  * TODO:Let the user decide, on which host the DB runs.
 18  * 
 19  * @constructor
 20  * @param {Function} param_cb
 21  * @api public
 22  */
 23 
 24 function MongoAdapter (param_cb) {
 25 	var that = this;
 26 
 27 	if (!this.db) {
 28 		this.db = new mongo.Db('wsload', new mongo.Server('localhost',mongo.Connection.DEFAULT_PORT, {}), {});
 29 		
 30 		this.db.open(function(err, p_db) {
 31 			if (err) {
 32 				param_cb(err);
 33 			} else {
 34 				param_cb(null,that.db);
 35 			}
 36 		});
 37 	} else {
 38 		param_cb(null,this.db);
 39 	}
 40 };
 41 
 42 /**
 43  * Closes the connection to the Database
 44  *
 45  * @param {Boolean} force Force disconnection
 46  * @param {Function} param_cb
 47  * @api public
 48  */
 49 
 50 MongoAdapter.prototype.close = function (param_force, param_cb) {
 51 	if (this.db) {
 52 		this.db.close(param_force,param_cb);
 53 	};
 54 };
