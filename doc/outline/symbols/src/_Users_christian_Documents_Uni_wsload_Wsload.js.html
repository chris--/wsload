 18 /**
 19  * Initialize a new Wsload object
 20  * 
 21  * @constructor
 22  * @param {Object} param_settings
 23  * @api public
 24  */
 25 
 26 function Wsload (param_settings) {
 27 	events.EventEmitter.call(this);
 28 
 29 	if(cluster.isMaster) {
 30 		if(!this.uuid) {
 31 			this.uuid = this._generateUuid();	
 32 		};
 33 		if (param_settings) {
 34 			this.logTarget = param_settings.logTarget;
 35 			if(param_settings.uuid) {
 36 				this.uuid = param_settings.uuid;
 37 			};
 38 		};
 39 	};
 40 };

 55 Wsload.prototype._generateUuid = function () {
 56 	//random uuid generator from https://gist.github.com/1308368
 57 	if (cluster.isMaster) { //only the master process is allowed to change the uuid
 58 		var uuid = function uuidGenerator(a,b){for(b=a='';a++<36;b+=a*51&52?(a^15?8^Math.random()*(a^20?16:4):4).toString(16):'-');return b;}
 59 		return uuid();
 60 	} 
 61 };

 63 /**
 64  * runs all suites
 65  *
 66  * @param {String} param_suiteName
 67  * @param {Integer} param_timesToRunSuite
 68  * @param {Array} param_testFunction
 69  * @param {Integer} param_suiteTimeout
 70  * @param {Object} param_globalUserVar
 71  * @api public
 72  */
 73 
 74 Wsload.prototype.runSuite = function (param_suiteName, param_timesToRunSuite, param_testFunctions, param_preTestFunction, param_suiteTimeout, param_globalVar) {
 75 	var workerCount = 0;
 76 	var that = this;
 77 
 78 	if (numCPUs>param_timesToRunSuite) {
 79 		//there are more cores than tests to run
 80 		workerCount = param_timesToRunSuite;
 81 	} else {
 82 		workerCount = numCPUs;
 83 	};
 84 
 85 	if (cluster.isMaster) {
 86 		// Fork workers.
 87 		for (var i = 0; i < workerCount; i++) {
 88 			var worker = cluster.fork({uuid:this.uuid, workerCount:workerCount, workerId:i, logTarget:this.logTarget});
 89 			worker.on('message', function (event) {
 90 				that.emit(event.type, event.msg)
 91 			});
 92 		};
 93 		
 94 		cluster.on('exit', function (worker, code, signal) {
 95 			if (!--workerCount) {
 96 				that._tearDown();
 97 			}
 98 		});
 99 	} else {
100 		//calculate 
101 		this._spawnWorker(param_suiteName, param_timesToRunSuite, param_testFunctions, param_preTestFunction, param_suiteTimeout, param_globalVar);
102 	};
103 };

105 /**
106  * emit "finished" event, this gets called when all tests are completed
107  *
108  * @api private
109  */
110 
111 Wsload.prototype._tearDown = function () {
112 	this._computeResult();
113 	this.emit('finished');
114 };

116 /**
117  * Spawns a Worker with help of the Cluster module
118  *
119  * @param {String} param_suiteName
120  * @param {Integer} param_timesToRunSuite
121  * @param {Array} param_testFunction
122  * @param {Integer} param_suiteTimeout
123  * @param {Object} param_globalUserVar
124  * @api private
125  */
126 
127 Wsload.prototype._spawnWorker = function (param_suiteName, param_timesToRunSuite, param_testFunctions, param_preTestFunction, param_suiteTimeout, param_globalVar) {
128 	//result array
129 	var results = [];
130 
131 	//run on the cluster
132 	var Testsuite = require('./lib/Testsuite.js');
133 	
134 
135 	//parse Parameter
136 	this.uuid = process.env.uuid;
137 	var workerId = parseInt(process.env.workerId);
138 	var timesToRunSuite = parseInt(param_timesToRunSuite);
139 	var workerCount = parseInt(process.env.workerCount);
140 
141 	//calculate Testsuite start# for this worker, check github issue#2
142 	var start = (workerId * timesToRunSuite)/workerCount;
143 	start = Math.round(start);
144 
145 	//calculate Testsuite stop# for this worker, check github issue#2
146 	var stop = (((workerId+1) * timesToRunSuite)/workerCount) - 1;
147 	stop = Math.round(stop);
148 	//DEBUG Print: console.log('id: ' + process.env.workerId + ', start: ' + Math.round(start) + ', stop: ' + Math.round(stop));
149 
150 	//keep track of finished suites
151 	var suitesFinished = start;
152 	var suitesCreated = start;
153 
154 	//times to run suite per core
155 	var timesToRunSuitePerCore = stop-start+1;
156 
157 	//create suites
158 	var suites = [];
159 	while (timesToRunSuitePerCore--) {
160 		suites.push(new Testsuite(this.uuid, suitesCreated++, param_suiteName, param_suiteTimeout, param_testFunctions, param_preTestFunction, param_globalVar));
161 	}
162 
163 	var that = this;
164 	//add listeners
165 	suites.forEach (function (testsuite) {
166 		testsuite.on('finished', function (result) {
167 			results.push(result);
168 			suitesFinished++;
169 			if(suitesFinished === suitesCreated) {
170 				//we are done with this Worker, send 
171 				var logger = new Logger({logTarget:process.env.logTarget});
172 				logger.log(results, function (err) {
173 					if(err) throw new Error(err);
174 					logger.closeDb();
175 					var clusterInstance = cluster.worker;
176 					clusterInstance.destroy();
177 				});
178 				
179 			};
180 		});
181 
182 		testsuite.on('timeout', function (suite, test) {
183 			var msgString = 'Testsuite ' + param_suiteName + '#' + suite + ' timed out in testcase ' + test;
184 			process.send({
185 				type:'timeout', 
186 				msg: msgString
187 			});
188 		});
189 	});
190 	
191 	//run suites
192 	suites.forEach(function(testsuite){
193 		testsuite.run();
194 	});
195 };
